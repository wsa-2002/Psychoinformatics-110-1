def get_prev_html(now_html: lxml.html.HtmlElement) -> lxml.html.HtmlElement:
    """
    page_index: 
    """
    prev_html = now_html.xpath('//a[@class="btn wide"]')[1].get("href")
    base_url = 'http://www.ptt.cc'
    r = urllib.request.Request(base_url + prev_html, headers={'User-Agent': ''})
    data = urllib.request.urlopen(r).read()
    t = lxml.html.fromstring(data.decode('utf-8'))
    return t
